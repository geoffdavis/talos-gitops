Alternative: Browser Extension

For a more permanent solution, you can create a browser extension that automatically fills the token. Save this as a userscript for Tampermonkey or similar:

// ==UserScript==
// @name         Kubernetes Dashboard Auto-Login
// @namespace    http://k8s.home.geoffdavis.com/
// @version      1.0
// @description  Auto-fill Kubernetes Dashboard token
// @match        https://dashboard.k8s.home.geoffdavis.com/*
// @grant        GM_setValue
// @grant        GM_getValue
// ==/UserScript==

(function() {
    'use strict';
    
    // Store your token here or fetch it from a secure location
    const TOKEN = GM_getValue('dashboardToken', '');
    
    // Function to fill and submit token
    function autoLogin() {
        const tokenInput = document.querySelector('input[type="password"], input[name="token"]');
        const submitButton = document.querySelector('button[type="submit"]');
        
        if (tokenInput && submitButton && TOKEN) {
            tokenInput.value = TOKEN;
            tokenInput.dispatchEvent(new Event('input', { bubbles: true }));
            setTimeout(() => submitButton.click(), 500);
        }
    }
    
    // Wait for page to load and try auto-login
    setTimeout(autoLogin, 1000);
    
    // Also watch for changes in case of SPA navigation
    const observer = new MutationObserver(() => {
        if (document.querySelector('input[type="password"]')) {
            autoLogin();
        }
    });
    observer.observe(document.body, { childList: true, subtree: true });
})();
